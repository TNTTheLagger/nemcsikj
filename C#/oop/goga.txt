#include <string>

};
struct A0 {
A0() { cout << '1'; }
~A0() { cout << 'd'; }
};
class Alap : public A0 {
int al;
string n;
public:
Alap(int i, const char *p = "N123OO") : n(p) {
if (i == 0) throw Hiba(1);
al = i;
cout << i << 'k';
}
Alap(const Alap& a) :al(a.al) { cout << al << 'C' << endl << endl;}
Alap& operator=(const Alap& a) { al = a.al; cout << a.n; return *this; }
~Alap() { cout << 'D'; }
};

Alap a0(52); cout << endl;
